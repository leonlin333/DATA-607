pluck_chr <- function(x, ...) {
  val <- purrr::pluck(x, ..., .default = NULL)
  if (is.null(val) || length(val) == 0) NA_character_ else as.character(val[[1]])
}

laureates_raw_tbl <- tibble(raw = laur_raw$laureates)

# ── Bio level (one row per laureate) ─────────────────────────────────────────
bio <- laureates_raw_tbl |>
  hoist(raw,
    id          = "id",
    known_name  = list("knownName", "en"),
    full_name   = list("fullName",  "en"),
    org_name    = list("orgName",   "en"),
    gender      = "gender",
    birth_date  = list("birth", "date"),
    birth_ctry  = list("birth", "place", "country",    "en"),
    birth_ctry2 = list("birth", "place", "countryNow", "en"),
    prizes      = "nobelPrizes"
  ) |>
  mutate(
    name          = coalesce(known_name, full_name, org_name),
    gender        = coalesce(gender, "organization"),
    birth_year    = as.integer(str_extract(coalesce(birth_date, ""), "^\\d{4}")),
    birth_country = coalesce(birth_ctry, birth_ctry2)
  ) |>
  select(id, name, gender, birth_year, birth_country, prizes)

# ── Prize level (one row per laureate × prize) ───────────────────────────────
prizes_per_person <- bio |>
  select(id, birth_year, birth_country, gender, prizes) |>
  unnest_longer(prizes) |>
  mutate(
    award_year  = as.integer(pluck_chr(prizes, "awardYear")),
    category    = map_chr(prizes, ~ pluck_chr(.x, "category", "en")),
    aff_country = map_chr(prizes, function(p) {
      affs <- p$affiliations
      if (is.null(affs) || length(affs) == 0) return(NA_character_)
      a <- affs[[1]]
      coalesce(pluck_chr(a, "country", "en"),
               pluck_chr(a, "countryNow", "en"))
    })
  ) |>
  select(id, birth_year, birth_country, gender,
         award_year, category, aff_country)

Answer: The United States, Germany, and the United Kingdom lead by birth country. However, many US-affiliated winners were born abroad and emigrated — the US is a net recipient of scientific talent. Germany’s high birth count reflects its 19th-century educational dominance, while its post-1933 exodus is the subject of Question 2.

Question 2 — Which country has the most “brain drain”? (Join across fields)

Question: A laureate born in Country A but affiliated with an institution in Country B at the time of their award represents a loss of intellectual capital from Country A. Which birth countries lost the most laureates this way?

This requires comparing birth_country and aff_country within the same laureate record — a cross-field filter that cannot be answered from either field alone.

drain <- prizes_per_person |>
  filter(
    !is.na(birth_country), birth_country != "",
    !is.na(aff_country),   aff_country   != "",
    birth_country != aff_country
  ) |>
  distinct(id, birth_country, aff_country)

emigrated <- drain |>

Answer: Germany has lost the most Nobel laureates in absolute numbers. A large share were Jewish scientists who fled after 1933 and received prizes at US and UK universities. Hungary and Poland show the highest rates — over 50% of their born laureates won prizes while working abroad, reflecting World War II displacement and Cold War political persecution.

Question 3 — How has the average age at award changed over the decades?

Question: Is there a widening gap between scientific discovery and Nobel recognition? If so, laureates should be older on average in recent decades than they were a century ago.

Answer: The average age at award has risen by roughly 12–15 years since 1901. In the early decades, science laureates were typically in their mid-50s; today they are usually in their late 60s. The Economics prize (established 1969) has the oldest average laureates. Literature and Peace are more stable — those prizes recognize sustained careers or advocacy rather than a single, dateable experimental discovery.

Question 4 — How often are prizes shared, and is collaboration increasing over time?

Question: Nobel science prizes can be shared among up to three people. If modern science is more collaborative, we should see the share of three-way splits rising over time and solo prizes declining.

share_data <- prize_tbl |>
  filter(!is.na(n_laureates), n_laureates > 0) |>
  mutate(
    share_type = case_when(
      n_laureates == 1 ~ "Solo",
      n_laureates == 2 ~ "Split (2)",
      n_laureates == 3 ~ "Split (3)",
      TRUE             ~ "Other"
    ),
    decade = (award_year %/% 10) * 10
  )

share_by_cat <- share_data |>
  count(category, share_type) |>
  group_by(category) |>
  mutate(pct = round(100 * n / sum(n), 1)) |>
  ungroup()
