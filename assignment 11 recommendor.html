cosine_sim <- function(a, b) {
  common <- !is.na(a) & !is.na(b)
  if (sum(common) < 2) return(NA_real_)       
  num   <- sum(a[common] * b[common])
  denom <- sqrt(sum(a[common]^2)) * sqrt(sum(b[common]^2))
  if (denom == 0) return(0)
  num / denom
}

predict_rating <- function(user, item, sim_mat, train_mat,
                           user_means, K = 3) {
  u_idx <- which(rownames(train_mat) == user)
  i_idx <- which(colnames(train_mat) == item)

  sims       <- sim_mat[u_idx, ]
  rated_item <- !is.na(train_mat[, i_idx])
  rated_item[u_idx] <- FALSE          

  eligible           <- sims * rated_item
  eligible[is.na(eligible)] <- 0

  
  if (sum(eligible > 0) == 0) return(user_means[u_idx])

  top_k <- order(eligible, decreasing = TRUE)[1:min(K, sum(eligible > 0))]
  top_k <- top_k[eligible[top_k] > 0]

  wts  <- sims[top_k]
  devs <- train_mat[top_k, i_idx] - user_means[top_k]

  pred <- user_means[u_idx] + sum(wts * devs) / sum(abs(wts))
  max(1, min(5, pred))   
}
top_n_recs <- function(user, n = 3, train_mat, sim_mat,
                       user_means, K = 3) {
  u_idx   <- which(rownames(train_mat) == user)
  unrated <- which(is.na(train_mat[u_idx, ]))
  if (length(unrated) == 0) return(data.frame())

  preds <- sapply(names(unrated), function(item) {
    predict_rating(user, item, sim_mat, train_mat, user_means, K)
  })

  df <- data.frame(
    User             = user,
    Item             = names(preds),
    Predicted_Rating = round(preds, 2),
    stringsAsFactors = FALSE
  )
  df[order(-df$Predicted_Rating)[1:min(n, nrow(df))], ]
}
